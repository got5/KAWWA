001package net.atos.kawwaportal.components.mixins;
002
003import org.apache.tapestry5.BindingConstants;
004import org.apache.tapestry5.ClientElement;
005import org.apache.tapestry5.annotations.AfterRender;
006import org.apache.tapestry5.annotations.Import;
007import org.apache.tapestry5.annotations.InjectContainer;
008import org.apache.tapestry5.annotations.Parameter;
009import org.apache.tapestry5.ioc.annotations.Inject;
010import org.apache.tapestry5.ioc.annotations.Symbol;
011import org.apache.tapestry5.json.JSONObject;
012import org.apache.tapestry5.services.javascript.JavaScriptSupport;
013import org.got5.tapestry5.jquery.JQuerySymbolConstants;
014
015/**
016 * 
017 * @tapestrydoc
018 *
019 */
020@Import(library = "classpath:/net/atos/kawwaportal/components/assets/readMore.js")
021public class ReadMore {
022        
023        @Parameter(value = "message:read-more-header", defaultPrefix = BindingConstants.MESSAGE)
024        private String header;
025        
026        @InjectContainer
027        private ClientElement clientElement;
028        
029    
030        @Inject
031        @Symbol(JQuerySymbolConstants.JQUERY_ALIAS)
032        private String jqueryAlias;
033        
034        @Inject 
035        private JavaScriptSupport javascriptSupport;
036        
037        @AfterRender
038        void afterRender() {
039                JSONObject params = new JSONObject();
040                params.accumulate("header", header);
041                String init = null;
042                init = String.format("%s('#%s').readMore(%s);", jqueryAlias, clientElement.getClientId(),params);
043                javascriptSupport.addScript(init);              
044        }
045
046}
